CREATE TRIGGER before_insert_students
BEFORE INSERT ON students
FOR EACH ROW
BEGIN
    DECLARE student_count INT;
    
    SELECT COUNT(*)
    INTO student_count
    FROM students
    WHERE name = NEW.name;

    IF student_count > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Duplicate student name is not allowed';
    END IF;
END //

DELIMITER ;
